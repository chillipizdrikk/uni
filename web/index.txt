    <!-- Форма входу -->
    <h2>Login</h2>
    <form id="loginForm">
        <label for="emailLogin">Email:</label>
        <input type="email" id="emailLogin" name="email" required>
        <label for="passwordLogin">Password:</label>
        <input type="password" id="passwordLogin" name="password" required>
        <button type="submit">Login</button>
    </form>

    <div id="taskContainer"></div>

    <script>
        const socket = io();
        let authToken = '';

        // Реєстрація користувача
        $('#registerForm').submit(function (event) {
            event.preventDefault();
            const userData = {
                username: $('#username').val(),
                email: $('#email').val(),
                password: $('#password').val()
            };
            $.post('/api/auth/register', userData, function (data) {
                alert('User registered successfully');
            }).fail(function (response) {
                alert('Failed to register: ' + response.responseJSON.error);
            });
        });

        // Вхід користувача
        $('#loginForm').submit(function (event) {
            event.preventDefault();
            const loginData = {
                email: $('#emailLogin').val(),
                password: $('#passwordLogin').val()
            };
            $.post('/api/auth/login', loginData, function (data) {
                authToken = data.token;
                alert('Logged in successfully');
            }).fail(function (response) {
                alert('Failed to login: ' + response.responseJSON.error);
            });
        });

        $('#taskForm').submit(function (event) {
            event.preventDefault();
            const number = $('#number').val();
            $.ajax({
                url: '/api/tasks/factorial',
                method: 'POST',
                headers: {
                    'Authorization': `Bearer ${authToken}`
                },
                data: { number },
                success: function (data) {
                    const taskId = data.taskId;
                    const progressBar = `<div id="task-${taskId}" class="progress-bar">
                                            <div class="progress-bar-fill" id="fill-${taskId}">0%</div>
                                         </div>`;
                    $('#taskContainer').append(progressBar);
                    socket.emit('startTask', taskId);
                }
            });
        });

        socket.on('progressUpdate', function (data) {
            const { taskId, progress } = data;
            const fill = $(`#fill-${taskId}`);
            fill.css('width', `${progress}%`).text(`${progress}%`);
        });

        socket.on('taskComplete', function (data) {
            const { taskId, result } = data;
            const fill = $(`#fill-${taskId}`);
            fill.css('width', '100%').text(`Complete: ${result}`);
        });
    </script>
